import pandas as pd

# –ü—Ä–∏–º–µ—Ä–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ

bootstrap = BootstrapABTest(
    data=df,
    metric="revenue",
    group_column="group",
    group_names=["control", "test"],
    n=10000,
    alpha=0.05,
    two_sided=True,
    random_state=42
)

bootstrap.run(show_plots=True)
bootstrap.summary()

# –ü–æ–ª—É—á–∏—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –≤—Ä—É—á–Ω—É—é:
